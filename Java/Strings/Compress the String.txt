        String result = "";

		   c++;	   
	    }
	    else {

		   if(c != 1)
		   result = result + inputString.charAt(i) + c;
		   else
		   result = result + inputString.charAt(i);
           c = 1;
	    }
    }
        if(c != 1)
        result = result + inputString.charAt(i) + c;
        else
        result = result + inputString.charAt(i);   
        return result;
    }
	
	public static void main(String[] args) {

		Scanner s = new Scanner(System.in);

        System.out.print(result);
