public class Solution {  

    public static void sortZeroesAndOne(int[] arr) {

        int total=0;
        for (int i=0;i<arr.length;i++) {

            total=total+arr[i];
        }
        for (int i=0;i<arr.length;i++) {

            if (i<arr.length-total) {

                arr[i]=0;
            }
            else
            {
                arr[i]=1;
            }   
        }
    }
}
